
\frame{
	\frametitle{Shiny: Analyses $\rightarrow$ Interactive Web Applications}

	Preliminaries 
	\begin{itemize}
		\item Shiny is an R package that makes it easy to build apps
		\item \texttt{install.packages("shiny")}
	\end{itemize}

	Running a Shiny app
	\begin{itemize}
		\item If Shiny app is in a directory called \texttt{my-app}:
		\begin{enumerate}
			\item library(shiny)
			\item runApp(`my-app')
		\end{enumerate}
		\item R \textbf{session} will be busy while Shiny app is active
		\item Get R session back $\rightarrow$ hit escape
	\end{itemize}
